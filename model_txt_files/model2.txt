model
{
  for (i in 1:n) {
    log(mu[i]) <- inprod(x[i, ], beta[])  # Assuming beta[] is a row vector
    Y[i] ~ dgamma(a1[i], a2[i])
    a1[i] <- phi
    a2[i] <- phi / mu[i]
  }
  
  precision_matrix[1:p, 1:p] <- inverse(0.001 * I)  # Assuming p is the dimension of beta
  beta[1:p] ~ dmnorm(rep(0, p), precision_matrix)

  phiinv ~ dgamma(2, 1)
  phi <- 1 / phiinv
}
